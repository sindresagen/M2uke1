                    let isFound = false;

(Stigende) s.sort(); Når vi sorterer en liste av objekter, er det umulig for JS å vite hva den skal sortere etter. Da må du sende med en såkal sammenligningsfunksjon (Compare function) sammenligningsfunksjonen skal ta to parametere og svare 0 hvis de er like, positivt tall hvis den første er sist - og negativt tall hvis den andre er sist. function getSum(myArray) {
                    let sum = 0;
                    for (let index = 0; index < myArray.length; index++) {
                        const number = myArray[index];
                        sum += number;
                    }  
                    return sum;
